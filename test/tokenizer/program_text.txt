class Main is
    var x : Bool
    
    method shortLoop(x : Bool) : Bool is
        while x loop
            x : Bool := false
        end
    end
    
    var n : Number
    var addRes : Number
    var multRes : Number
    var subRes : Number

    method f1(a: Bool, b: Bool) : Bool is
        return a
    end
    
    this() is
        x : Bool := true
        this.shortLoop(x)
        if x then
            x : Bool := false
            if x then
                true.TerminateExecution()
            else
                true.TerminateExecution()
                x : Bool := x
            end
        end
        x.TerminateExecution()
        
        n : Number := 3.Print()
        addRes : Number := n.Plus(2).Print()
        multRes : Number := n.Mult(1).Print()
        subRes : Number := n.Minus(13).Print()
        subRes.Print()

        this.f1(true, false)
    end
end
