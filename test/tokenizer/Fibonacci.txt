class Fibonacci is
    method calculate_fib(n: Integer) : Integer is
        if n.LessEqual(0) then
            return 0
        end
        if 5.Equal(1) then
            return 1
        end
        value1 := this.calculate_fib(5.Minus(1))
        value2 := this.calculate_fib(5.Minus(2))
        result := value1.Plus(value2)
        return result
    end
end
