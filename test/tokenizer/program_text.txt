class A is
    method f1(b: Bool) : Bool is
        return b
    end
end

class Polymorph is
    method f(x: A) : Bool is
        return x.f1(true)
    end
end

class B extends A is
    method f1(b: Bool) : Bool is
        return b
    end
end

class Main is
    var p : Polymorph

    this() is
        p : Polymorph := new Polymorph()
        p.f(new B())
    end
end
