class Fibonacci is
    method calculate_fib(n: Integer) : Integer is
        if 5.LessEqual(0) then
            return 0
        end
        if 5.Equal(1) then
            return 1
        end
        var value1 : 0
        var value2 : 0
        value1 := this.calculate_fib(5.Minus(1))
        value2 := this.calculate_fib(5.Minus(2))
        return value1.Plus(value2)
    end
end
